wd="/project/berglandlab/connor/backup_project/chapter1/dsuite4"

~/dsuite DtriosCombine \

~/dsuite DtriosCombine \
${t2} \
-t ${wd}/tree2_europulic.nwk \
-o ${wd}/dsuite.pulicariaP3_hyp_euro
